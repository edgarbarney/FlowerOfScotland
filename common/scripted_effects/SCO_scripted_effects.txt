SCO_civil_war_finished = {
	# In all ways
	SCO = {
		add_ideas = volunteer_only
		country_lock_all_division_template = no
		set_country_flag = SCO_civilwar_over
		set_country_flag = after_first_independence_war
		#white_peace = ENG
		remove_ideas = SCO_idea_for_freedom
	}
	
	if = {
		limit = {
			SCO = {
				has_completed_focus = SCO_focus_dominanceway
			}
		}
		
		SCO = {
			country_event = { id = foscot.flagchange.0 days = 1 }
		}
		
		every_state = {
			if = {
				limit = {
					is_core_of = SCO
				}
				remove_core_of = ENG
			}
			if = {
				limit = {
					is_core_of = WLS
					OR = {
						is_owned_by = WLS
						is_owned_by = ENG
					}
				}
				remove_core_of = ENG
				add_core_of = SCO
			}
			if = {
				limit = {
					is_core_of = NIR
					OR = {
						is_owned_by = NIR
						is_owned_by = ENG
					}
				}
				remove_core_of = ENG
				add_core_of = SCO
			}
			if = {
				limit = {
					is_core_of = ENG
				}
				add_core_of = SCO
			}
		}

		SCO = {
			country_event = {
				id = foscot.fallofunion.0
				days = 2
				random_days = 1
				random_hours = 16
			}
		}
	}
}
